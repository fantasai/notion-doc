% function tables
\newlength{\fndw}
\newlength{\descw}
\settowidth{\descw}{\begin{tabular}{ll}Description: & \end{tabular}}
\setlength{\fndw}{\textwidth}
\addtolength{\fndw}{-\descw}

\newenvironment{function}{
  \begin{tabular*}{\textwidth}{rp{\fndw}}
}{
  \\
  \end{tabular*}
  \par
}
\newcommand{\synopsis}[1]{
  Synopsis: & \code{#1} \\
  Description: &
}
